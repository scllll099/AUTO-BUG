local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local correctKey = "Halisbestfr"  -- ✅ Buraya kendi keyini yaz
local attempts = 0

-- UI oluştur
local ScreenGui = Instance.new("ScreenGui", LocalPlayer:WaitForChild("PlayerGui"))
local Frame = Instance.new("Frame", ScreenGui)
Frame.Size = UDim2.new(0, 350, 0, 180)
Frame.Position = UDim2.new(0.5, -175, 0.5, -90)
Frame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
Frame.Active = true
Frame.Draggable = true

local TextBox = Instance.new("TextBox", Frame)
TextBox.Size = UDim2.new(0, 250, 0, 40)
TextBox.Position = UDim2.new(0.5, -125, 0, 30)
TextBox.PlaceholderText = "Key Giriniz..."
TextBox.Text = ""
TextBox.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
TextBox.TextColor3 = Color3.fromRGB(255,255,255)

local Button = Instance.new("TextButton", Frame)
Button.Size = UDim2.new(0, 120, 0, 40)
Button.Position = UDim2.new(0.5, -60, 0, 90)
Button.Text = "Doğrula"
Button.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
Button.TextColor3 = Color3.fromRGB(255,255,255)

local Info = Instance.new("TextLabel", Frame)
Info.Size = UDim2.new(1, 0, 0, 30)
Info.Position = UDim2.new(0,0,1,-30)
Info.BackgroundTransparency = 1
Info.TextColor3 = Color3.fromRGB(255,0,0)
Info.Text = ""

-- Key kontrol fonksiyonu
local function checkKey()
    if TextBox.Text == correctKey then
        ScreenGui:Destroy()
        loadScript() -- ✅ Doğru key → script çalışacak
    else
        attempts += 1
        Info.Text = "Yanlış Key! ("..attempts.."/3)"
        if attempts >= 3 then
            LocalPlayer:Kick("3 defa yanlış key girdiniz!")
        end
    end
end

Button.MouseButton1Click:Connect(checkKey)

--// KEY SYSTEM BİTİŞ \\--


--// ASIL SCRIPT BAŞLANGIÇ \\--
function loadScript()
    local RunService = game:GetService("RunService")
    local killRadius = 25 
    local sphereColor = Color3.fromRGB(255, 50, 50)
    local sphereTransparency = 0.8
    _G.AutoKillSphere = false
    _G.whitelistedPlayers = {}

    local toolName = "Rock"
    _G.RockFarm = false
    _G.FastPunch = false

    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local hrp = character:WaitForChild("HumanoidRootPart")
    local humanoid = character:WaitForChild("Humanoid")

    local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/memejames/elerium-v2-ui-library/main/Library"))()
    local window = library:AddWindow("SECRET SCRİPT PUBLİC | by halis  ", {
        main_color = Color3.fromRGB(255, 40, 40),
        min_size = Vector2.new(400, 400),
        toggle_key = Enum.KeyCode.RightControl
    })

    local mainTab = window:AddTab("Ana Kontrol")
    local infoTab = window:AddTab("Bilgi")

    infoTab:AddLabel("BU SCRİPT PUBLİC VERSYONDUR VE YAKINDA SİLİNCEK")
    infoTab:AddLabel("PAİD VERSYON ÇOK DAHA GELİŞMİŞ HALDE OLACAK")
    infoTab:AddLabel("")
    infoTab:AddLabel("KİLL SPHERE SAYESİNDE SİZE YAKLAŞAN")
    infoTab:AddLabel("OTOMATİK ÖLECEKTİR KİMSE SİZİ ÖLDÜREMEZ")
    infoTab:AddLabel("")
    infoTab:AddLabel("24 saate ping ve fps'inize bağlı olarak")
    infoTab:AddLabel("8.1-7.2m arası bug yapabilirsiniz")
    infoTab:AddLabel("")
    infoTab:AddLabel("Paid versiyon ücreti 5USD olması planlanıyor")
    infoTab:AddLabel("")
    infoTab:AddLabel("by halis")

    LocalPlayer.CharacterAdded:Connect(function(newChar)
        character = newChar
        hrp = character:WaitForChild("HumanoidRootPart")
        humanoid = character:WaitForChild("Humanoid")
        
        if _G.RockFarm then
            task.wait(1)
            equipTool()
            equipPunch()
        end
        
        if _G.AutoKillSphere then
            task.wait(1)
            startKillLoop()
        end
    end)

    local function equipTool()
        local backpack = LocalPlayer:WaitForChild("Backpack")
        local tool = backpack:FindFirstChild(toolName) or character:FindFirstChild(toolName)
        if tool and tool.Parent ~= character then
            tool.Parent = character
            humanoid:EquipTool(tool)
        end
    end

    local function equipPunch()
        local backpack = LocalPlayer:WaitForChild("Backpack")
        local punchTool = backpack:FindFirstChild("Punch") or character:FindFirstChild("Punch")
        if punchTool and punchTool.Parent ~= character then
            punchTool.Parent = character
            humanoid:EquipTool(punchTool)
        end
    end

    local teleportLoop
    local function startTeleportLoop()
        if teleportLoop then
            teleportLoop:Disconnect()
            teleportLoop = nil
        end
        
        teleportLoop = RunService.Heartbeat:Connect(function()
            if _G.RockFarm and hrp and hrp.Parent then
                hrp.CFrame = CFrame.new(-8979.12, 9.19, -6130.21) * CFrame.new(0, 0, 11)
            end
        end)
    end

    local punchLoop
    local function startPunchLoop()
        if punchLoop then
            punchLoop:Disconnect()
            punchLoop = nil
        end
        
        punchLoop = RunService.Heartbeat:Connect(function()
            if _G.FastPunch then
                equipPunch()
                
                if LocalPlayer:FindFirstChild("muscleEvent") then
                    LocalPlayer.muscleEvent:FireServer("punch", "leftHand")
                    LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                end
            end
        end)
    end

    local toolLoop
    local function startToolLoop()
        if toolLoop then
            toolLoop:Disconnect()
            toolLoop = nil
        end
        
        toolLoop = RunService.Heartbeat:Connect(function()
            if _G.RockFarm then
                equipTool()
            end
        end)
    end

    local function createKillSphere()
        if _G.KillSphere then _G.KillSphere:Destroy() end
        
        local sphere = Instance.new("Part")
        sphere.Shape = Enum.PartType.Ball
        sphere.Size = Vector3.new(killRadius*2, killRadius*2, killRadius*2)
        sphere.Transparency = sphereTransparency
        sphere.Color = sphereColor
        sphere.Anchored = true
        sphere.CanCollide = false
        sphere.Material = Enum.Material.Neon
        sphere.Name = "BUG_SC_KillSphere"
        
        local pointLight = Instance.new("PointLight")
        pointLight.Brightness = 0.5
        pointLight.Range = killRadius
        pointLight.Color = sphereColor
        pointLight.Parent = sphere
        
        _G.SphereUpdate = RunService.Heartbeat:Connect(function()
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                sphere.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame
            end
        end)
        
        sphere.Parent = workspace
        _G.KillSphere = sphere
        return sphere
    end

    local function killPlayer(target)
        if not LocalPlayer.Character then return end
        
        local leftHand = LocalPlayer.Character:FindFirstChild("LeftHand")
        if leftHand and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
            firetouchinterest(target.Character.HumanoidRootPart, leftHand, 0)
            firetouchinterest(target.Character.HumanoidRootPart, leftHand, 1)
            
            for _, v in pairs(LocalPlayer.Backpack:GetChildren()) do
                if v.Name == "Punch" then
                    LocalPlayer.Character.Humanoid:EquipTool(v)
                end
            end
            
            LocalPlayer.muscleEvent:FireServer("punch", "leftHand")
            LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
        end
    end

    local function startKillLoop()
        local sphere = createKillSphere()
        
        while _G.AutoKillSphere and task.wait(0.1) do
            if not LocalPlayer.Character then continue end
            
            local spherePos = sphere.Position
            for _, player in ipairs(Players:GetPlayers()) do
                if player == LocalPlayer then continue end
                if _G.whitelistedPlayers and table.find(_G.whitelistedPlayers, player.Name) then continue end
                
                if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                    local distance = (player.Character.HumanoidRootPart.Position - spherePos).Magnitude
                    if distance <= killRadius then
                        killPlayer(player)
                    end
                end
            end
        end
        
        if sphere then sphere:Destroy() end
        if _G.SphereUpdate then _G.SphereUpdate:Disconnect() end
    end

    local rockFolder = mainTab:AddFolder("Rock Farm")
    rockFolder:AddSwitch("AUTO ROCK", function(state)
        _G.RockFarm = state
        _G.FastPunch = state
        
        if state then
            equipTool()
            equipPunch()
            startTeleportLoop()
            startPunchLoop()
            startToolLoop()
        else
            if teleportLoop then teleportLoop:Disconnect() end
            if punchLoop then punchLoop:Disconnect() end
            if toolLoop then toolLoop:Disconnect() end
        end
    end)

    local killFolder = mainTab:AddFolder("Kill Sphere")
    killFolder:AddTextBox("Küre Boyutu (Stud)", function(text)
        local newSize = tonumber(text)
        if newSize and newSize > 0 then
            killRadius = newSize
            if _G.KillSphere then
                _G.KillSphere.Size = Vector3.new(killRadius*2, killRadius*2, killRadius*2)
                if _G.KillSphere:FindFirstChildOfClass("PointLight") then
                    _G.KillSphere:FindFirstChildOfClass("PointLight").Range = killRadius
                end
            end
        end
    end)

    killFolder:AddSwitch("Kill Sphere ", function(state)
        _G.AutoKillSphere = state
        if state then
            task.spawn(startKillLoop)
        else
            if _G.KillSphere then _G.KillSphere:Destroy() end
            if _G.SphereUpdate then _G.SphereUpdate:Disconnect() end
        end
    end)

    killFolder:AddColorpicker("Küre Rengi", sphereColor, function(color)
        sphereColor = color
        if _G.KillSphere then
            _G.KillSphere.Color = color
            if _G.KillSphere:FindFirstChildOfClass("PointLight") then
                _G.KillSphere:FindFirstChildOfClass("PointLight").Color = color
            end
        end
    end)

    killFolder:AddTextBox("Whitelist Ekle (İsim)", function(name)
        if name and name ~= "" then
            _G.whitelistedPlayers = _G.whitelistedPlayers or {}
            table.insert(_G.whitelistedPlayers, name)
        end
    end)

    killFolder:AddButton("Whitelist Temizle", function()
        _G.whitelistedPlayers = {}
    end)

    print("ByHalis.")
end
--// ASIL SCRIPT BİTİŞ \\--
